"use client";
import React, { useState, useEffect, useRef } from "react";
import { motion } from "framer-motion";
import { Card, CardContent } from "@/components/ui/card";
import { Dialog, DialogContent, DialogHeader, DialogTitle } from "@/components/ui/dialog";
import { Button } from "@/components/ui/button";
import { Volume2, VolumeX } from "lucide-react";

type Mood = {
  name: string;
  video: string;
  audio: string;
  tagline: string;
  perfumes: { name: string; notes: string }[];
};

const moods: Mood[] = [
  {
    name: "Beach",
    video: "/videos/mood-beach.mp4",
    audio: "/audio/beach-waves.mp3",
    tagline: "Fresh. Breezy. Ocean Escape.",
    perfumes: [
      { name: "Azure Mist", notes: "Citrus, Sea Salt, Musk" },
      { name: "Ocean Drift", notes: "Bergamot, Marine, Amber" },
    ],
  },
  {
    name: "Garden",
    video: "/videos/mood-garden.mp4",
    audio: "/audio/garden-birds.mp3",
    tagline: "Bloom with Nature.",
    perfumes: [
      { name: "Floral Serenade", notes: "Rose, Jasmine, Sandalwood" },
      { name: "Green Whisper", notes: "Herbs, Lily, Vetiver" },
    ],
  },
  {
    name: "Caf√©",
    video: "/videos/mood-cafe.mp4",
    audio: "/audio/cafe-ambience.mp3",
    tagline: "Warm. Cozy. Romantic.",
    perfumes: [
      { name: "Vanilla Noir", notes: "Vanilla, Coffee, Patchouli" },
      { name: "Spice Affair", notes: "Cinnamon, Clove, Tonka Bean" },
    ],
  },
];

export default function MoodExplorer2() {
  const [selectedMood, setSelectedMood] = useState<Mood | null>(null);
  const [isPlaying, setIsPlaying] = useState(true);
  const audioRef = useRef<HTMLAudioElement | null>(null);

  // Handle ambient sound playback
  useEffect(() => {
    if (selectedMood) {
      audioRef.current = new Audio(selectedMood.audio);
      audioRef.current.loop = true;
      audioRef.current.volume = 0.5;

      if (isPlaying) {
        audioRef.current.play().catch(() => {
          console.warn("Autoplay blocked. User interaction needed.");
        });
      }
    }
    return () => {
      if (audioRef.current) {
        audioRef.current.pause();
        audioRef.current = null;
      }
    };
  }, [selectedMood]);

  // Handle toggle play/pause
  useEffect(() => {
    if (audioRef.current) {
      if (isPlaying) {
        audioRef.current.play();
      } else {
        audioRef.current.pause();
      }
    }
  }, [isPlaying]);

  return (
    <div className="py-16 bg-white text-black">
      <h2 className="text-4xl font-bold text-center mb-10">Explore by Mood</h2>
      <div className="grid grid-cols-1 md:grid-cols-3 gap-6 px-6 max-w-6xl mx-auto">
        {moods.map((mood, i) => (
          <motion.div
            key={i}
            whileHover={{ scale: 1.05 }}
            className="cursor-pointer relative"
            onClick={() => {
              setSelectedMood(mood);
              setIsPlaying(true);
            }}
          >
            <Card className="relative overflow-hidden rounded-2xl shadow-lg h-64">
              {/* Video Background */}
              <video
                src={mood.video}
                className="absolute top-0 left-0 w-full h-full object-cover"
                autoPlay
                loop
                muted
                playsInline
              />
              {/* Overlay with mood info */}
              <CardContent className="absolute inset-0 bg-black/40 flex flex-col justify-center items-center text-center">
                <h3 className="text-2xl font-semibold">{mood.name}</h3>
                <p className="text-sm mt-2">{mood.tagline}</p>
              </CardContent>
            </Card>
          </motion.div>
        ))}
      </div>

      {/* Mood Modal */}
      <Dialog open={!!selectedMood} onOpenChange={() => setSelectedMood(null)}>
        <DialogContent className="bg-white text-black rounded-2xl p-6 max-w-lg relative">
          <DialogHeader>
            <DialogTitle className="text-2xl font-bold flex items-center justify-between">
              {selectedMood?.name} Mood Picks
              {/* Sound Toggle Button */}
              <Button
                variant="ghost"
                size="icon"
                onClick={() => setIsPlaying(!isPlaying)}
              >
                {isPlaying ? (
                  <Volume2 className="h-6 w-6 text-gray-700" />
                ) : (
                  <VolumeX className="h-6 w-6 text-gray-700" />
                )}
              </Button>
            </DialogTitle>
          </DialogHeader>
          <div className="mt-4 space-y-4">
            {selectedMood?.perfumes.map((perfume, i) => (
              <div key={i} className="p-3 border rounded-lg shadow-sm">
                <h4 className="text-lg font-semibold">{perfume.name}</h4>
                <p className="text-sm text-gray-600">Notes: {perfume.notes}</p>
                <Button className="mt-2 w-full">Try Now</Button>
              </div>
            ))}
          </div>
        </DialogContent>
      </Dialog>
    </div>
  );
}
